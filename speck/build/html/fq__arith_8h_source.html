106static inline
107FQ_ELEM fq_cond_sub(const FQ_ELEM x) {
108 // equivalent to: (x >= Q) ? (x - Q) : x
109 // likely to be ~ constant-time (a "smart" compiler might turn this into conditionals though)
110 FQ_ELEM sub_q = x - Q;
111 FQ_ELEM mask = -(sub_q >> NUM_BITS_Q);
112 return (mask & Q) + sub_q;
113}

115static inline
116FQ_ELEM fq_red_barrett(const FQ_DOUBLEPREC x) {
117 // Barrett reduction for Q = 127 (full reduction as long as: x < 129 * 127)
118 // mu = ceil((1<<16) / 127) = 517
119 // t1 = (mu * x) / (2 ^ 16)
120 // This function is likely to be ~ constant-time
121 uint16_t t1 = (((uint32_t) x << 9) + ((uint32_t) x << 2) + x) >> 16;
122 uint16_t t2 = x - t1;
123 t1 += (t2 >> 1);
124 t1 >>= 6;
125 t1 -= (t1 << 7);
126 return x + t1;
127}

183FQ_ELEM fq_pow(FQ_ELEM x, FQ_ELEM exp) {
184 FQ_DOUBLEPREC xlift;
