int8_t ab = (int8_t)b ^ (int8_t)a; \
int8_t c = (int8_t)b - (int8_t)a; \
c ^= ab & (c ^ (int8_t)b); \
c >>= 7; \
c &= ab; \
a ^= (FQ_ELEM)c; \
b ^= (FQ_ELEM)c; \
} while(0)
