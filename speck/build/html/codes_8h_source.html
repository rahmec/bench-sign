126void row_mat_mult(FQ_ELEM *out,
127 const FQ_ELEM *vec,
128 const FQ_ELEM M[K][K_pad],
129 uint8_t r,
130 uint8_t c){
131
132 vec256_t shuffle, t, c7f, c01, b, a, a_lo, a_hi, b_lo, b_hi;
133 vec128_t tmp;
134 vec256_t res;
135 vset8(c7f, 127);
136 vset8(c01, 1);
137
138 for (uint32_t col = 0; (col+32) <= NEXT_MULTIPLE(c, 32); col+=32) {
139 // precompute
140
141 vset8(res, 0);
142 vload256(shuffle, (vec256_t *) shuff_low_half);
143 //vset8(c7f, 127);
144 //vset8(c01, 1);
145
146 for (uint32_t row = 0; row < r; row+=1){
147 // precompute b
148 vset8(b, vec[row]);
149 vget_lo(tmp, b);
150 vextend8_16(b_lo, tmp);
151 vget_hi(tmp, b);
152 vextend8_16(b_hi, tmp);
153
154 vload256(a, (vec256_t *)(M[row]+col));
155
156 vget_lo(tmp, a);
157 vextend8_16(a_lo, tmp);
158 vget_hi(tmp, a);
159 vextend8_16(a_hi, tmp);
160
161 barrett_mul_u16(a_lo, a_lo, b_lo, t);
162 barrett_mul_u16(a_hi, a_hi, b_hi, t);
163
164 vshuffle8(a_lo, a_lo, shuffle);
165 vshuffle8(a_hi, a_hi, shuffle);
166
167 vpermute_4x64(a_lo, a_lo, 0xd8);
168 vpermute_4x64(a_hi, a_hi, 0xd8);
169
170 vpermute2(t, a_lo, a_hi, 0x20);
171
172 // barrett_red8(t, r, c7f, c01);
173 W_RED127_(t);
174 vadd8(res,res,t)
175 W_RED127_(res);
176 }
177
178 vstore256((vec256_t *)(out + col), res);
179 }
