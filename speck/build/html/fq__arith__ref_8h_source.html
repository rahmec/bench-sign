105static inline
106FQ_ELEM fq_cond_sub(const FQ_ELEM x) {
107 // equivalent to: (x >= Q) ? (x - Q) : x
108 // likely to be ~ constant-time (a "smart" compiler might turn this into conditionals though)
109 FQ_ELEM sub_q = x - Q;
110 FQ_ELEM mask = -(sub_q >> NUM_BITS_Q);
111 return (mask & Q) + sub_q;
112}

114static inline
115FQ_ELEM fq_red_barrett(const FQ_DOUBLEPREC x) {
116 // Barrett reduction for Q = 127 (full reduction as long as: x < 129 * 127)
117 // mu = ceil((1<<16) / 127) = 517
118 // t1 = (mu * x) / (2 ^ 16)
119 // This function is likely to be ~ constant-time
120 uint16_t t1 = (((uint32_t) x << 9) + ((uint32_t) x << 2) + x) >> 16;
121 uint16_t t2 = x - t1;
122 t1 += (t2 >> 1);
123 t1 >>= 6;
124 t1 -= (t1 << 7);
125 return x + t1;
126}

181static inline
182FQ_ELEM fq_pow(FQ_ELEM x, FQ_ELEM exp) {
183 FQ_DOUBLEPREC xlift;
184 xlift = x;
185 FQ_DOUBLEPREC accum = 1;
186 /* No need for square and mult always, Q-2 is public*/
187 while(exp) {
188 if(exp & 1) {
189 accum = fq_red(accum*xlift);
190 }
191 xlift = fq_red(xlift*xlift);
192 exp >>= 1;
193 }
194 return fq_red(accum);
195} /* end fq_pow */
