500void bitonic_sort_i8(FQ_ELEM *x,
501 const long long n) {
502 long long p, q, r, i;
503 if (n < 2) {
504 return;
505 }
506 const long long top = 1ul << (32 - __builtin_clz(K/2));
507
508 for (p = top; p > 0; p >>= 1) {
509 for (i = 0; i < n - p; ++i) {
510 if (!(i & p)) {
511 int8_MINMAX(x[i], x[i + p]);
512 }
513 }
514
515 i = 0;
516 for (q = top; q > p; q >>= 1) {
517 for (; i < n - q; ++i) {
518 if (!(i & p)) {
519 FQ_ELEM a = x[i + p];
520 for (r = q; r > p; r >>= 1) {
521 int8_MINMAX(a, x[i + r]);
522 }
523
524 x[i + p] = a;
525 }
526 }
527 }

556int row_bitonic_sort(normalized_IS_t *G) {
557 // first sort each row into a tmp buffer
558 FQ_ELEM tmp[K][Q] __attribute__((aligned(32)));
559 FQ_ELEM* ptr[K] __attribute__((aligned(32)));
560 uint32_t P[K];
561 for (uint32_t i = 0; i < K; ++i) {
562 sort(tmp[i], G->values[i], N-K);
563
564 ptr[i] = tmp[i];
565 P[i] = i;
566 }
567
568 const uint64_t n = K;
569 const uint64_t top = 1ul << (32 - __builtin_clz(K/2));
570
571 for (uint64_t p = top; p > 0; p >>= 1) {
572 for (uint64_t i = 0; i < n - p; ++i) {
573 if (!(i & p)) {
574 // NOTE: here is a sign cast, this is needed, so the
575 // sign extension needed for the mask is an unsigned one.
576 const int32_t cmp1 = compare_rows_bitonic_sort(ptr, i, i + p);
577 const uint32_t cmp = (uint32_t)cmp1;
578 if (cmp == 0) { return 0; }
579
580 const uintptr_t mask = -(1ull - (cmp >> 31));
581 cswap((uintptr_t *)(&ptr[i]), (uintptr_t *)(&ptr[i+p]), mask);
582 MASKED_SWAP(P[i], P[i+p], mask);
583 }
584 }
585
586 for (uint64_t q = top; q > p; q >>= 1) {
587 for (uint64_t i = 0; i < n - q; ++i) {
588 if (!(i & p)) {
589 for (uint64_t r = q; r > p; r >>= 1) {
590 const uint32_t cmp = (uint32_t)compare_rows_bitonic_sort(ptr, i+p, i + r);
591 if (cmp == 0) { return 0; }
592
593 const uintptr_t mask = -(1ull - (cmp >> 31));
594 cswap((uintptr_t *)(&ptr[i+p]), (uintptr_t *)(&ptr[i+r]), mask);
595 MASKED_SWAP(P[i+p], P[i+r], mask);
596 }
597 }
598 }
599 }
600 }
601
602 // apply the permutation
603 for (uint32_t t = 0; t < K; t++) {
604 uint32_t ind = P[t];
605 while(ind<t) { ind = P[ind]; }
606
607 normalized_row_swap(G, t, ind);
608 }
609
610 return 1;
611}

770 uint32_t P[K];
771 for (uint32_t i = 0; i < K; ++i) {
772 ptr[i] = G->values[i];
773 P[i] = i;
774 }
775
776 uint64_t n = K;
777
778 const uint64_t top = 1ul << (32 - __builtin_clz(K/2));
779
780 for (uint64_t p = top; p > 0; p >>= 1) {
781 for (uint64_t i = 0; i < n - p; ++i) {
782 if (!(i & p)) {
783 // NOTE: here is a sign cast, this is needed, so the
784 // sign extension needed for the mask is an unsigned one.
785 const uint32_t cmp = (uint32_t)compare_rows_bitonic_sort(ptr, i, i + p);
786
787 const uintptr_t mask = -(1ull - (cmp >> 31));
788 cswap((uintptr_t *)(&ptr[i]), (uintptr_t *)(&ptr[i+p]), mask);
789 MASKED_SWAP(P[i], P[i+p], mask);
790 }
791 }
792
793 for (uint64_t q = top; q > p; q >>= 1) {
794 for (uint64_t i = 0; i < n - q; ++i) {
795 if (!(i & p)) {
796 for (uint64_t r = q; r > p; r >>= 1) {
797 const uint32_t cmp = (uint32_t)compare_rows_bitonic_sort(ptr, i+p, i + r);
798 const uintptr_t mask = -(1ull - (cmp >> 31));
799 cswap((uintptr_t *)(&ptr[i+p]), (uintptr_t *)(&ptr[i+r]), mask);
800 MASKED_SWAP(P[i+p], P[i+r], mask);
801 }
802 }
803 }
804 }
805 }
806
807 // apply the permutation
808 for (uint32_t t = 0; t < K; t++) {
809 uint32_t ind = P[t];
810 while(ind<t) { ind = P[ind]; }
811
812 normalized_row_swap(G, t, ind);
813 }
814
815 return 1;
816}
